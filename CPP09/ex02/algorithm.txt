função ford_johnson(container, order):

    total_elementos = tamanho(container)
    par_size        = order * 2
    grupo_count     = total_elementos / par_size

    // Caso base: não dá pra formar nem um par de grupos
    se grupo_count < 1:
        retorna

    // Verificar se sobra um grupo sem par (straggler)
    tem_straggler = (total_elementos / order) % 2 != 0


    // -----------------------------------------------
    // DESCIDA: ordenar cada par de grupos pelo big
    // -----------------------------------------------
    para i de 0 até grupo_count - 1:

        idx_a = i * par_size           // início do grupo A (big)
        idx_b = i * par_size + order   // início do grupo B (small)

        big_a = container[idx_a + order - 1]
        big_b = container[idx_b + order - 1]

        se big_b > big_a:
            swap_grupos(container, idx_a, idx_b, order)

    // Recursão: desce pro próximo nível
    ford_johnson(container, order * 2)


    // -----------------------------------------------
    // SUBIDA: montar main e pend, inserir com jacobsthal
    // -----------------------------------------------

    // Extrair grupos A (big) como main chain
    // Extrair grupos B (small) como pend
    // Cada elemento de main e pend é um grupo de tamanho 'order'
    main = []
    pend = []

    para i de 0 até grupo_count - 1:
        idx_a = i * par_size
        idx_b = i * par_size + order
        main.adicionar( container[idx_a .. idx_a + order - 1] )
        pend.adicionar( container[idx_b .. idx_b + order - 1] )

    // Se tem straggler, adiciona ao pend
    se tem_straggler:
        idx_straggler = grupo_count * par_size
        pend.adicionar( container[idx_straggler .. idx_straggler + order - 1] )

    // b1 é garantidamente menor que a1, inserir no início da main
    main.inserir_no_inicio( pend[0] )
    pend.remover(0)

    // Gerar ordem de inserção baseada em Jacobsthal
    jacobsthal_order = gerar_indices_jacobsthal(tamanho(pend))

    // Inserir cada grupo do pend na main
    para cada índice i em jacobsthal_order:

        grupo  = pend[i]
        big    = grupo[order - 1]   // maior elemento do grupo

        // O parceiro A de pend[i] está em main
        // pend[i] é garantidamente menor que seu parceiro A
        // então o limite superior da busca é a posição do parceiro A
        limite = posição_do_parceiro_a(main, i, order)

        posição = binary_search(main, big, de=0, até=limite)

        main.inserir(grupo, posição)

    container = main
